34 // 2D rectangle type with helper methods.
35 template<typename T>
36 struct XU_API Bounds2 {
37  Point2<T> lower, upper;
38 
39  constexpr Bounds2() = default;
40  constexpr Bounds2(T left, T right, T top, T bottom) :
41  lower(left, top),
42  upper(right, bottom) {}
43  constexpr Bounds2(const Point2<T>& lower, const Point2<T>& upper) :
44  lower(lower),
45  upper(upper) {}
46  constexpr Bounds2(const std::initializer_list<Point2<T>>& vertices) :
47  lower(std::numeric_limits<T>::max(), std::numeric_limits<T>::max()),
48  upper(std::numeric_limits<T>::lowest(),
49  std::numeric_limits<T>::lowest()) {
50  for (const auto& v : vertices) {
51  lower.x = std::min(v.x, lower.x);
52  lower.y = std::min(v.y, lower.y);
53  upper.x = std::max(v.x, upper.x);
54  upper.y = std::max(v.y, upper.y);
55  }
56  }
57 
58  constexpr bool ContainsPoint(const Point2<T>& p) const {
59  return p >= lower && p <= upper;
60  }
61 
62  constexpr bool Overlaps(const Bounds2& other) const {
63  if (upper.x < other.lower.x || lower.x > other.upper.x) return false;
64  if (upper.y < other.lower.y || lower.y > other.upper.y) return false;
65 
66  return true;
67  }
68 
69  constexpr bool FullyContains(const Bounds2& other) const {
70  return lower <= other.lower && upper >= other.upper;
71  }
72 
73  // Returns a new Bounds2<T> with the size of (size * (1 + factor))
74  constexpr Bounds2<T> Fattened(float factor) const {
75  if constexpr (std::is_signed_v<T>)
76  assert(factor >= -1.f); // Attempts to fatten past -1 will result in
77  // an invalid bounding box
78 
79  Vector2<T> deltaSize;
80  factor /= 2.f; // Div by two to properly follow the equation size = size
81  // + size*factor
82 
83  // Overload for floating points to make sure factor is the same type as
84  // T
85  if constexpr (std::is_floating_point_v<T>)
86  deltaSize = (upper - lower) * static_cast<T>(factor);
87  else // If T is not floating point, ensure that multipication is proper
88  // and then truncate down
89  deltaSize = static_cast<Vector2<T>>(
90  static_cast<FVector2>((upper - lower)) * factor);
91 
92  return Bounds2<T>(lower - deltaSize, upper + deltaSize);
93  }
94 
95  constexpr Vector2<T> Size() const { return upper - lower; }
96 
97  constexpr T& Top() { return lower.y; }
98  constexpr const T& Top() const { return lower.y; }
99 
100  constexpr T& Bottom() { return upper.y; }
101  constexpr const T& Bottom() const { return upper.y; }
102 
103  constexpr T& Left() { return lower.x; }
104  constexpr const T& Left() const { return lower.x; }
105 
106  constexpr T& Right() { return upper.x; }
107  constexpr const T& Right() const { return upper.x; }
108 
109  bool operator==(const Bounds2<T>& rhs) const {
110  return lower == rhs.lower && upper == rhs.upper;
111  }
112 
113  bool operator!=(const Bounds2<T>& rhs) const { return !(*this == rhs); }
114 
115  template<typename Cast_t>
116  constexpr operator Bounds2<Cast_t>() const {
117  return Bounds2<Cast_t>(static_cast<Point2<Cast_t>>(lower),
118  static_cast<Point2<Cast_t>>(upper));
119  }
120 };
